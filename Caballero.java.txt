public class Caballero extends Soldado {
    private boolean montado;
    private String arma;

    public Caballero(String nombre, int puntosVida, int fila, int columna, boolean montado) {
        super(nombre, puntosVida, fila, columna);
        this.montado = montado;
        this.arma = montado ? "lanza" : "espada";
    }

    public void montar() {
        if (!montado) {
            montado = true;
            arma = "lanza";
            System.out.println(nombre + " monta y prepara su " + arma + ".");
        }
    }

    public void desmontar() {
        if (montado) {
            montado = false;
            arma = "espada";
            System.out.println(nombre + " desmonta y cambia su arma a " + arma + ".");
        }
    }

    public void envestir() {
        int ataques = montado ? 3 : 2;
        System.out.println(nombre + " realiza una embestida con " + ataques + " ataques.");
    }

    @Override
    public void mostrarAccion() {
        envestir();
    }

    @Override
    public String toString() {
        return super.toString() + String.format(", Montado: %b, Arma: %s", montado, arma);
    }
}
