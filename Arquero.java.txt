public class Arquero extends Soldado {
    private int flechasDisponibles;

    public Arquero(String nombre, int puntosVida, int fila, int columna, int flechasDisponibles) {
        super(nombre, puntosVida, fila, columna);
        this.flechasDisponibles = flechasDisponibles;
    }

    public void dispararFlecha() {
        if (flechasDisponibles > 0) {
            flechasDisponibles--;
            System.out.println(nombre + " dispara una flecha. Flechas restantes: " + flechasDisponibles);
        } else {
            System.out.println(nombre + " no tiene flechas disponibles.");
        }
    }

    @Override
    public void mostrarAccion() {
        dispararFlecha();
    }

    @Override
    public String toString() {
        return super.toString() + String.format(", Flechas Disponibles: %d", flechasDisponibles);
    }
}
